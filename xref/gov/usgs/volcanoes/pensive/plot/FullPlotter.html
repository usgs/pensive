7   package gov.usgs.volcanoes.pensive.plot;
8   
9   import gov.usgs.plot.render.TextRenderer;
10  import gov.usgs.plot.render.wave.SliceWaveRenderer;
11  import gov.usgs.plot.render.wave.SpectrogramRenderer;
12  import gov.usgs.volcanoes.core.configfile.ConfigFile;
13  
14  import java.awt.Dimension;
15  import java.awt.Font;
16  
17  
18  /**
19   * A concreate class to produce a large image with decorations.
20   * 
21   * @author Tom Parker
22   */
23  public class FullPlotter extends ChannelPlotter {
24  
25    /** Font used to indicate that no data was available. */
26    public static final Font NO_DATA_FONT = Font.decode("dialog-PLAIN-36");
27  
28    /** height of plot decorations in pixels. */
29    public static final int LABEL_HEIGHT = 35;
30  
31    /** width of plot decorations in pixels. */
32    public static final int LABEL_WIDTH = 30;
33  
34    /**
35     * Class constructor.
36     * 
37     * @param name My name
38     * @param index My position on the subnet plot
39     * @param plotDimension The dimension of my plot
40     * @param decorateX If true, decorate the X-axis
41     * @param config My configuration stanza
42     */
43    public FullPlotter(String name, int index, Dimension plotDimension, boolean decorateX,
44        ConfigFile config) {
45      super(name, index, plotDimension, config);
46  
47      spectrogramRenderer.xTickValues = decorateX;
48      spectrogramRenderer.xUnits = decorateX;
49      spectrogramRenderer.xLabel = decorateX;
50  
51      noDataFont = NO_DATA_FONT;
52    }
53  
54    /**
55     * Apply any settings needed to my SpectrogramRenderer.
56     * 
57     * @param my SpectrogramRenderer
58     */
59    protected void tweakSpectrogramRenderer(SpectrogramRenderer spectrogramRenderer) {
60      spectrogramRenderer.yTickMarks = true;
61      spectrogramRenderer.yTickValues = true;
62      spectrogramRenderer.xTickMarks = true;
63      spectrogramRenderer.setYLabelText(name);
64  
65      int top = index * plotDimension.height;
66      spectrogramRenderer.setLocation(LABEL_WIDTH, top + waveHeight + LABEL_HEIGHT,
67          plotDimension.width - (2 * LABEL_WIDTH), plotDimension.height - waveHeight);
68    }
69  
70    /**
71     * Apply any settings needed to my WaveRenderer.
72     * 
73     * @param my WaveRenderer
74     */
75    protected void tweakWaveRenderer(SliceWaveRenderer waveRenderer) {
76      int top = index * plotDimension.height + LABEL_HEIGHT;
77      int width = plotDimension.width;
78      waveRenderer.setLocation(LABEL_WIDTH, top, width - 2 * LABEL_WIDTH, waveHeight);
79  
80      waveRenderer.xTickMarks = true;
81    }
82  
83    /**
84     * Apply any settings needed to my TextRenderer.
85     * 
86     * @param my TextRenderer
87     */
88    protected void tweakNoDataRenderer(TextRenderer textRenderer) {
89      textRenderer.y += LABEL_HEIGHT;
90    }
91  
92  }
