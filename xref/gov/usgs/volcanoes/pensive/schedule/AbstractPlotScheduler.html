7   package gov.usgs.volcanoes.pensive.schedule;
8   
9   import gov.usgs.volcanoes.core.configfile.ConfigFile;
10  import gov.usgs.volcanoes.pensive.PlotJob;
11  import gov.usgs.volcanoes.pensive.WaveSource;
12  import gov.usgs.volcanoes.pensive.plot.SubnetPlotter;
13  
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  import java.util.ArrayList;
18  import java.util.LinkedList;
19  import java.util.List;
20  import java.util.concurrent.BlockingQueue;
21  import java.util.concurrent.PriorityBlockingQueue;
22  
23  /**
24   * Create a pool of connections to a single server and assign plot jobs to
25   * connections as they become available.
26   *
27   * @author Tom Parker
28   */
29  public abstract class AbstractPlotScheduler implements Runnable {
30  
31    /** my logger. */
32    protected static final Logger LOGGER = LoggerFactory.getLogger("gov.usgs");
33  
34    /** number of concurrent connection to the wave server. */
35    public static final int DEFAULT_NUMTHREADS = 5;
36  
37    /** pool of plotters, each with their own wave server connection. */
38    private List<WaveSource> waveSources;
39  
40    private final List<Thread> threads;
41  
42    /** Queue of plot jobs awaiting an available plotter. */
43    protected final BlockingQueue<PlotJob> plotJobs;
44  
45    /** list of subnets that feed from my wave server. */
46    protected final List<SubnetPlotter> subnets;
47  
48    /** number of connections to my wave server. */
49    private final int numThreads;
50  
51    /** name of this server. */
52    public final String name;
53  
54    private final ConfigFile config;
55  
56    protected abstract void schedulePlots();
57  
58    /**
59     * Class constructor.
60     *
61     * @param name name given to this wave server in the config file
62     * @param config My configuration stanza
63     */
64    public AbstractPlotScheduler(final String name, final ConfigFile config) {
65  
66      this.name = name;
67      numThreads = config.getInt("threads", DEFAULT_NUMTHREADS);
68      subnets = new LinkedList<SubnetPlotter>();
69      plotJobs = new PriorityBlockingQueue<PlotJob>();
70      threads = new LinkedList<Thread>();
71      this.config = config;
72    }
73  
74    /**
75     * Start threads.
76     */
77    public void startWaveSources() {
78      waveSources = new ArrayList<WaveSource>();
79      for (int i = 0; i < numThreads; i++) {
80        final String n = name + "-" + i;
81        final WaveSource ws = new WaveSource(n, plotJobs, config);
82        waveSources.add(ws);
83        final Thread t = new Thread(ws);
84        t.setName(n);
85        t.start();
86        threads.add(t);
87      }
88  
89    }
90  
91    /**
92     * Add a subnet.
93     * @param subnet
94     *          The subnet to be added
95     */
96    public void add(final SubnetPlotter subnet) {
97      subnets.add(subnet);
98    }
99  
100   /**
101    * Return the count of subnets.
102    * 
103    * @return count of subnets I have
104    */
105   public int subnetCount() {
106     return subnets.size();
107   }
108 
109   /**
110    * Stop scheduling plots.
111    */
112   public void stop() {
113     for (final WaveSource ws : waveSources) {
114       ws.stop();
115     }
116   }
117 
118   /**
119    * Schedule the next set of plots. Try to catch all exceptions,
120    * ScheduledExecutorService does the wrong thing with exceptions.
121    */
122   public void run() {
123     try {
124       LOGGER.info("Scheduling plots for " + name);
125       schedulePlots();
126     } catch (final Exception e) {
127       LOGGER.error("Caught exception heading for scheduler. " + e.getLocalizedMessage());
128     }
129   }
130 }
